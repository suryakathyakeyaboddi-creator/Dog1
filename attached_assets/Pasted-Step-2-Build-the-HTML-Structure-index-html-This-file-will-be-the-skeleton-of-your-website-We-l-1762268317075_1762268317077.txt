This is where the magic happens! We'll use your JavaScript snippet to connect to the Gradio client.

Copy and paste this into your script.js file:

JavaScript

//
// 1. IMPORT THE GRADIO CLIENT
//
// We import the client directly from its CDN (Content Delivery Network) URL.
// This is possible because we set type="module" in our HTML <script> tag.
//
import { Client } from "https://cdn.jsdelivr.net/npm/@gradio/client/dist/index.min.js";

//
// 2. GET REFERENCES TO OUR HTML ELEMENTS
//
const fileInput = document.getElementById("image-upload");
const previewImg = document.getElementById("preview");
const identifyBtn = document.getElementById("identify-btn");
const loadingDiv = document.getElementById("loading");
const resultsDiv = document.getElementById("results");

// Store the uploaded file
let userImageFile = null;

//
// 3. EVENT LISTENER FOR FILE UPLOAD
//
fileInput.addEventListener("change", (event) => {
  if (event.target.files && event.target.files[0]) {
    userImageFile = event.target.files[0];

    // Show image preview
    const reader = new FileReader();
    reader.onload = (e) => {
      previewImg.src = e.target.result;
      previewImg.style.display = "block";
    };
    reader.readAsDataURL(userImageFile);

    // Clear previous results
    resultsDiv.classList.add("hidden");
  }
});

//
// 4. EVENT LISTENER FOR THE "IDENTIFY BREED" BUTTON
//
identifyBtn.addEventListener("click", async () => {
  if (!userImageFile) {
    alert("Please upload an image first!");
    return;
  }

  // Disable button and show loading spinner
  identifyBtn.disabled = true;
  identifyBtn.innerText = "Analyzing...";
  loadingDiv.classList.remove("hidden");
  resultsDiv.classList.add("hidden");

  try {
    //
    // 5. CONNECT TO YOUR HUGGING FACE API
    //
    const client = await Client.connect("SuryaKathyakeyaBoddi/dog-breed-ai");

    //
    // 6. SEND THE IMAGE AND GET THE RESULT
    //
    // The client can accept a File object directly (like our userImageFile)
    const result = await client.predict("/predict", {
      image: userImageFile,
    });

    // Log the full result to the console so you can inspect its structure
    console.log("API Result:", result.data);

    //
    // 7. DISPLAY THE RESULT
    //
    // We assume the API returns data in a format like:
    // [ { "label": "Golden Retriever", "conf": 0.95 }, { "label": "..." } ]
    // You may need to adjust this based on your API's *actual* output.
    //
    const topResult = result.data[0];
    const breedName = topResult.label;
    const confidence = (topResult.conf * 100).toFixed(1); // Format to one decimal place

    resultsDiv.innerHTML = `
      <h2>I'm ${confidence}% sure that's a...</h2>
      <h3>${breedName}</h3>
    `;
    
  } catch (error) {
    // Handle any errors
    console.error("API Call Failed:", error);
    resultsDiv.innerHTML = `<p style="color: red;">Error: Could not identify breed. Please try again.</p>`;
  } finally {
    //
    // 8. RE-ENABLE THE BUTTON AND HIDE LOADING
    //
    loadingDiv.classList.add("hidden");
    resultsDiv.classList.remove("hidden");
    identifyBtn.disabled = false;
    identifyBtn.innerText = "Identify Breed";
  }
});
